def is_number_palindrome(num):
    original_num = num
    reversed_num = 0

    while num > 0:
        digit = num % 10
        reversed_num = (reversed_num * 10) + digit
        num = num // 10

    if original_num == reversed_num:
        return True
    else:
        return False

num1 = 123
num2 = 456

if is_number_palindrome(num1):
    print(num1, "PASS")
else:
    print(num1, "FAIL")

if is_number_palindrome(num2):
    print(num2, "PASS")
else:
    print(num2, "FAIL")
